222  void finishNode() {
223  // Iff we ended up serializing an empty object or array, writeName
224  // will never have been called - therefore the node type at the top
225  // of the stack will be StartArray or StartObject - so start and
226  // then immediately end the documents/array.
227 
228  // We'll also end any documents/arrays we happen to be in.
229 
230  if (_nodeTypeStack.empty()) {
231  throw bson_mapper::Exception("Attempting to finish a nonexistent node.");
232  }
233 
234  switch (_nodeTypeStack.top()) {
235  case OutputNodeType::StartArray:
236  _bsonBuilder.open_array();
237  ++_arrayNestingLevel;
238  // Don't break so that this array can also be closed.
239  case OutputNodeType::InArray:
240  _bsonBuilder.close_array();
241  --_arrayNestingLevel;
242  break;
243  case OutputNodeType::StartObject:
244  // Only consider creating a new document if we're not in dot notation mode, or if we
245  // are within an array somewhere.
246  if (!_dotNotationMode || _arrayNestingLevel > 0) {
247  if (_nodeTypeStack.size() > 1 || _objAsRootElement) {
248  _bsonBuilder.open_document();
249  }
250  } else if (_nodeTypeStack.size() > 1) {
251  // Push back a dummy name so we don't accidentally pop an empty stack when
252  // closing this empty document.
253  _embeddedNameStack.push_back("");
254  }
255  // Don't break so that this document can also be closed.
256  case OutputNodeType::InObject:
257  // Only close the document if we are not in dot notation mode or if we are within an
258  // array somewhere.
259  if (!_dotNotationMode || _arrayNestingLevel > 0) {
260  if (_nodeTypeStack.size() > 1) {
261  _bsonBuilder.close_document();
262  } else if (_objAsRootElement) {
263  _bsonBuilder.close_document();
264  }
265  } else if (_nodeTypeStack.size() > 1) {
266  // Pop the name of this embedded document off the stack.
267  _embeddedNameStack.pop_back();
268  }
269  break;
270  }
271 
272  _nodeTypeStack.pop();
273  _curNodeInheritsUnderlyingBSONDataBase.pop();
274 
275  // Reset the _objAsRootElement tracker bool if the stack was emptied.
276  if (_nodeTypeStack.empty()) {
277  _objAsRootElement = false;
278  }
279  writeDocIfRoot();
280  }

351  void writeName(bool isNewNode = false) {
352  if (!_nodeTypeStack.empty()) {
353  const auto& topType = _nodeTypeStack.top();
354 
355  // Start up either an object or an array, depending on state.
356  if (topType == OutputNodeType::StartArray) {
357  _bsonBuilder.open_array();
358  ++_arrayNestingLevel;
359  _nodeTypeStack.top() = OutputNodeType::InArray;
360  } else if (topType == OutputNodeType::StartObject) {
361  _nodeTypeStack.top() = OutputNodeType::InObject;
362  // Only open a document if this document is not in the root node or this is a root
363  // element.
364  if (_nodeTypeStack.size() > 1 || _objAsRootElement) {
365  if (_dotNotationMode && _arrayNestingLevel == 0) {
366  _embeddedNameStack.push_back(_nextPotentialNodeName);
367  } else {
368  _bsonBuilder.open_document();
369  }
370  }
371  }
372 
373  // Elements in arrays do not have names.
374  if (topType == OutputNodeType::InArray) return;
375  }
376 
377  if (!_nextName && _nodeTypeStack.empty() && isNewNode) {
378  // This is a document at the root node with no name.
379  // Do nothing since no name is expected for non root-element documents in root.
380  return;
381  } else if (!_nextName) {
382  // Enforce the existence of a name-value pair if this is not a node in root,
383  // or this in an element in root.
384  throw bson_mapper::Exception("Missing a name for current node or element.");
385  } else {
386  // Set the key of this element to the name stored by the archiver.
387  if (!_dotNotationMode || _embeddedNameStack.empty() || _arrayNestingLevel > 0) {
388  _bsonBuilder.key_view(_nextName);
389  } else {
390  // If we are in dot notation mode and we're not nested in array, build the name of
391  // this key.
392  std::stringstream key;
393  for (const auto& name : _embeddedNameStack) {
394  key << name << ".";
395  }
396  key << _nextName;
397  _bsonBuilder.key_owned(key.str());
398  }
399 
400  // Save the name of this key in case it is the name of an embedded document.
401  _nextPotentialNodeName = _nextName;
402 
403  // Reset the name of the next key.
404  _nextName = nullptr;
405 
406  // If this a named node in the root node, set up this node as a root element.
407  if (_nodeTypeStack.empty() && isNewNode) {
408  _objAsRootElement = true;
409  }
410  }
411  }
412 
417  void makeArray() {
418  _nodeTypeStack.top() = OutputNodeType::StartArray;
419  }

422  // The BSONCXX builder for this archive.
423  BSONBuilder _bsonBuilder;
424 
425  // The stream to which to write the BSON archive.
426  std::ostream& _writeStream;
427 
428  // The name of the next element to be added to the archive.
429  char const* _nextName;

431  // The name of the last element added to the archive, which may potentially be the name of the
432  // next embedded document. This is stored to support dot notation mode.
433  char const* _nextPotentialNodeName;

470  BSONInputArchive(std::istream& stream)

472  _nextName(nullptr),
473  _readStream(stream),
474  _readFirstDoc(false) {

477  private:
482  void readNextDoc() {
483  // Determine the size of the BSON document in bytes.
484  // TODO: Only works on little endian.
485  int32_t docsize;
486  char docsize_buf[sizeof(docsize)];
487  _readStream.read(docsize_buf, sizeof(docsize));
488  std::memcpy(&docsize, docsize_buf, sizeof(docsize));
489 
490  // Throw an exception if the end of the stream is prematurely reached.
491  if (_readStream.eof() || !_readStream || docsize < 5) {
492  throw bson_mapper::Exception("No more data in BSONInputArchive stream.");
493  }
494 
495  // Create a shared buffer to store the BSON document.
496  _curBsonData =
497  std::shared_ptr<uint8_t>{new uint8_t[docsize], [](uint8_t* p) { delete[] p; }};
498 
499  // Read the BSON data from the stream into the buffer.
500  std::memcpy(_curBsonData.get(), docsize_buf, sizeof(docsize));
501  _readStream.read(reinterpret_cast<char*>(_curBsonData.get() + sizeof(docsize)),
502  docsize - sizeof(docsize));
503 
504  // Make sure there were no errors reading the BSON data.
505  if (_readStream.eof() || !_readStream) {
506  throw bson_mapper::Exception("No more data in BSONInputArchive stream.");
507  }
508 
509  // Store the BSON data of the document in a view that we can access.
510  _curBsonDoc = bsoncxx::document::view{_curBsonData.get(), static_cast<size_t>(docsize)};
511  _curBsonDataSize = docsize;
512 
513  // Specify that we've read a document.
514  _readFirstDoc = true;
515  }

523  inline bsoncxx::types::value search() {
524  // If our search result is cached, return the cached result instead of repeating the search.
525  if (_cachedSearchResult) {
526  auto val = *_cachedSearchResult;
527  _cachedSearchResult = stdx::nullopt;
528  return val;
529  }
530 
531  // Make sure the node stack is not empty.
532  if (_nodeTypeStack.empty()) {
533  throw bson_mapper::Exception("Cannot search for element if not in node.");
534  }
535 
536  // Check to make sure a document has been read.
537  if (!_readFirstDoc) {
538  throw bson_mapper::Exception(
539  "Error in element search, never loaded BSON data from file.");
540  }
541 
542  if (_nextName) {
543  // Reset _nextName
544  const char* nextName = _nextName;
545  _nextName = nullptr;
546 
547  if (_nodeTypeStack.top() == InputNodeType::InObject ||
548  _nodeTypeStack.top() == InputNodeType::InRootElement) {
549  // If we're in an object in the Root (InObject),
550  // look for the key in the current BSON view.
551  const auto& elemFromDoc = _curBsonDoc[nextName];
552  if (elemFromDoc) {
553  return elemFromDoc.get_value();
554  }
555  } else if (_nodeTypeStack.top() == InputNodeType::InEmbeddedObject) {
556  // If we're in an embedded object, look for the key in the object
557  // at the top of the embedded object stack.
558  const auto& elemFromDoc = _embeddedBsonDocStack.top()[nextName];
559  if (elemFromDoc) {
560  return elemFromDoc.get_value();
561  }
562  }
563 
564  // Provide an exception with an error message if the key is not found.
565  std::stringstream error_msg;
566  error_msg << "No element found with the key ";
567  error_msg << nextName;
568  error_msg << ".";
569  throw bson_mapper::Exception(error_msg.str());
570 
571  } else if (_nodeTypeStack.top() == InputNodeType::InEmbeddedArray) {
572  // If we're in an array (InEmbeddedArray), retrieve an element from
573  // the array iterator at the top of the stack, and increment it for
574  // the next retrieval.
575  auto& iter = _embeddedBsonArrayIteratorStack.top();
576  const auto elemFromArr = *iter;
577  ++iter;
578 
579  if (elemFromArr) {
580  return elemFromArr.get_value();
581  }
582 
583  throw bson_mapper::Exception(
584  "Invalid element found in array, or array is out of bounds.");
585  }
586 
587  throw bson_mapper::Exception("Missing name for element search.");
588  }
589 
590  public:
601  bool willSearchYieldValue() {
602  // Make sure the node stack is not empty.
603  if (_nodeTypeStack.empty()) {
604  throw bson_mapper::Exception("Cannot search for element if not in node.");
605  }
606 
607  // Check to make sure a document has been read.
608  if (!_readFirstDoc) {
609  throw bson_mapper::Exception(
610  "Error in element search, never loaded BSON data from file.");
611  }
612 
613  if (_nodeTypeStack.top() == InputNodeType::InEmbeddedArray) {
614  throw cereal::Exception(
615  "Should not be checking if search() will yield next value from an embedded array.");
616  }
617 
618  if (_nextName) {
619  const char* nextName = _nextName;
620  _nextName = nullptr;
621 
622  bsoncxx::document::element val{};
623 
624  if (_nodeTypeStack.top() == InputNodeType::InObject ||
625  _nodeTypeStack.top() == InputNodeType::InRootElement) {
626  val = _curBsonDoc[nextName];
627 
628  } else {
629  val = _embeddedBsonDocStack.top()[nextName];
630  }
631 
632  if (val) {
633  _cachedSearchResult = val.get_value();
634  return true;
635  }
636  }
637 
638  _cachedSearchResult = stdx::nullopt;
639  return false;
640  }

670  void startNode() {
671  if (_nodeTypeStack.empty()) {
672  // If we are in the root node, read in the next document from the stream,
673  // and update the state of the node we're currently in.
674  readNextDoc();
675 
676  // If there is a name, the new node is loading a BSON document or array as a root
677  // element.
678  if (_nextName) {
679  _nodeTypeStack.push(InputNodeType::InRootElement);
680  auto newNode = search();
681 
682  if (newNode.type() == bsoncxx::type::k_array) {
683  _embeddedBsonArrayStack.push(newNode.get_array().value);
684  _embeddedBsonArrayIteratorStack.push(_embeddedBsonArrayStack.top().begin());
685  _nodeTypeStack.push(InputNodeType::InEmbeddedArray);
686  } else if (newNode.type() == bsoncxx::type::k_document) {
687  _embeddedBsonDocStack.push(newNode.get_document().value);
688  _nodeTypeStack.push(InputNodeType::InEmbeddedObject);
689  } else {
690  throw bson_mapper::Exception("Node requested is neither document nor array.");
691  }
692  } else {
693  _nodeTypeStack.push(InputNodeType::InObject);
694  }
695  } else {
696  // If we're not in the root node, match the next key to an embedded document
697  // or array.
698  // From the BSON document we're currently in, fetch the value associated
699  // with this node and update the relevant stacks.
700  auto newNode = search();
701 
702  if (newNode.type() == bsoncxx::type::k_document) {
703  _embeddedBsonDocStack.push(newNode.get_document().value);
704  _nodeTypeStack.push(InputNodeType::InEmbeddedObject);
705  } else if (newNode.type() == bsoncxx::type::k_array) {
706  _embeddedBsonArrayStack.push(newNode.get_array().value);
707  _embeddedBsonArrayIteratorStack.push(_embeddedBsonArrayStack.top().begin());
708  _nodeTypeStack.push(InputNodeType::InEmbeddedArray);
709  } else {
710  throw bson_mapper::Exception("Node requested is neither document nor array.");
711  }
712  }
713  }
714 
719  void finishNode() {
720  // If we're in an embedded object or array, pop it from its respective
721  // stack(s).
722  if (_nodeTypeStack.top() == InputNodeType::InEmbeddedObject) {
723  _embeddedBsonDocStack.pop();
724  } else if (_nodeTypeStack.top() == InputNodeType::InEmbeddedArray) {
725  _embeddedBsonArrayStack.pop();
726  _embeddedBsonArrayIteratorStack.pop();
727  }
728 
729  // Pop the node type from the stack.
730  _nodeTypeStack.pop();
731 
732  if (!_nodeTypeStack.empty() && _nodeTypeStack.top() == InputNodeType::InRootElement) {
733  _nodeTypeStack.pop();
734  }
735  }

863  void loadUnderlyingDataForCurrentNode(UnderlyingBSONDataBase& underlyingData) {
864  if (_nodeTypeStack.empty()) {
865  throw bson_mapper::Exception("Cannot get data; not currently in a node.");
866  }
867 
868  switch (_nodeTypeStack.top()) {
869  case InputNodeType::InObject:
870  case InputNodeType::InRootElement:
871  underlyingData.setUnderlyingBSONData(std::shared_ptr<uint8_t>(_curBsonData),
872  _curBsonDataSize);
873  return;
874  case InputNodeType::InEmbeddedObject:
875  // Use the aliasing constructor of shared_ptr to build a reference pointing to the
876  // embedded document while reference counting the BSON doc as a whole.
877  underlyingData.setUnderlyingBSONData(
878  std::shared_ptr<uint8_t>(
879  _curBsonData, const_cast<uint8_t*>(_embeddedBsonDocStack.top().data())),
880  _embeddedBsonDocStack.top().length());
881  return;
882  case InputNodeType::InEmbeddedArray:
883  throw bson_mapper::Exception(
884  "Underlying BSON data does not support array views. Wrap the b_array, or your "
885  "container holding other bsoncxx view types in a class that inherits "
886  "bson_mapper::UnderlyingBSONDataBase");
887  return;
888  }
