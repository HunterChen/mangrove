856 constexpr std::enable_if_t<(N < M) && hasField<Base, T, N, M>::value, const nvp<Base, T>> wrapimpl(

859  return std::get<N>(Base::mongo_odm_mapped_fields());
860  } else {
861  return wrapimpl<Base, T, N + 1, M>(t);
862  }
863 }

868  T Base::*t) {

875  return nvp<Base, T>(nullptr, nullptr);
876 }
