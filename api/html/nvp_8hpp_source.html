147 // If Nth field has same type as T, check that it points to the same member.
148 // If not, check (N+1)th field.
149 template <typename Base, typename T, size_t N, size_t M>
150 constexpr std::enable_if_t<(N < M) && hasField<Base, T, N, M>::value, const Nvp<Base, T>> wrapimpl(
151  T Base::*t) {
